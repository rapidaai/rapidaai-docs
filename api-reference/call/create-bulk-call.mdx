---
title: "Create Bulk Phone Calls"
---

Creates multiple outbound phone calls in a single request using your configured assistant.

<Info>
  For full setup and authentication options, see the Installation guide: [Installation guide](/api-reference/installation) and [Authentication](/api-reference/authentication).
</Info>



## Parameters

  Configuration for the client connection.
</ParamField>

  <Expandable>
    <ParamField body="PhoneCalls" type="CreatePhoneCallRequest[]" required>
      <Expandable>
        <ParamField body="assistant" type="AssistantDefinition" required description="jf">
          <Expandable>
            <ParamField body="assistantId" type="string" required>
              AssistantID which will be used.

```ts React
import {   
  AssistantDefinition,

  ConnectionConfig,
  CreateBulkPhoneCallRequest,

const connectionCfg = ConnectionConfig.DefaultConnectionConfig(
  ConnectionConfig.WithSDK({ ApiKey: process.env.RAPIDA_API_KEY })

assistant.setAssistantid("ASSISTANT_ID_PLACEHOLDER");
assistant.setVersion("VERSION_PLACEHOLDER");
phoneCallRequest.setAssistant(assistant);
phoneCallRequest.setTonumber("TO_NUMBER_PLACEHOLDER");

    }
    let err = x.getError();
    if (err?.getHumanmessage()) setError(err?.getHumanmessage());
  })
  .catch((x) => {
    console.dir(x);
  });
```

```python Python
import os

    ConnectionConfig,
    create_bulk_phone_call,

connectionConfig = ConnectionConfig.default_connection_config(

)

    assistant=AssistantDefinition(assistantId={ASSISTANT_ID}),
    toNumber="{TO_PHONE}",
)
response = create_bulk_phone_call(

    request=CreateBulkPhoneCallRequest(
        phoneCalls=[call],
```

```go Go
package main

import (
	"context"
	"fmt"
	"os"

	clients "github.com/rapidaai/rapida-go/rapida/clients"
	lexatic_backend "github.com/rapidaai/rapida-go/rapida/clients/protos"
	connections "github.com/rapidaai/rapida-go/rapida/connections"
)

func main() {

	// for more authentication check https://doc.rapida.ai/api-reference/authentication
	connectionConfig := connections.
		DefaultconnectionConfig(connections.WithSDK(os.Getenv("RAPIDA_PROJECT_CREDENTIAL")))
	response, err := clients.CreateBulkPhoneCall(context.Background(), connectionConfig, &lexatic_backend.CreateBulkPhoneCallRequest{
		PhoneCalls: []*lexatic_backend.CreatePhoneCallRequest{
			// add multiple phone call request here to make call
			{
				Assistant: &lexatic_backend.AssistantDefinition{
					// change
					AssistantId: 2227823180112723968,
				},
				ToNumber: "{PHONE_NUMBER}",
			}},
	})

	if err != nil {
		fmt.Printf("error while making call using rapida %+v", err)
		return
	}
	fmt.Printf("rapida calling response %+v", response)
}
```
